// npm comes pre-installed with Node.js
// Check npm version
// npm --version

// ---------------- LOCAL DEPENDENCY ----------------
// Installed inside node_modules of your project.
// You use it by requiring/importing in your code.
// Example: Express (web framework)
// npm install express
//
// Usage in code:
// const express = require('express');
// const app = express();

// ---------------- GLOBAL DEPENDENCY ----------------
// Installed globally and available from anywhere on your system.
// Usually used for CLI tools (not inside code).
// Example: Nodemon (auto-restart server on changes)
// npm install -g nodemon
//
// Usage in terminal:
// nodemon server.js

// ---------------- MAC / LINUX NOTE ----------------
// On Mac/Linux, you may need to use 'sudo' for global install
// sudo npm install -g <package-name>
